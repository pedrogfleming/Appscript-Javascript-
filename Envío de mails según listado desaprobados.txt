function getLastRowSpecial(range)
{
  var rowNum = 0;
  var blank = false;
  for(var row = 0; row < range.length; row++)
  { 
    if(range[row][0] === "" && !blank)
    {
      rowNum = row;
      blank = true;
 
    }else if(range[row][0] !== "")
    {
      blank = false;
    }
  }
  return rowNum;
}
